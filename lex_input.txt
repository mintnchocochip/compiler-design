#include <iostream>
using namespace std;

class Calculator {
private:
    int result;
    
public:
    int add(int a, int b) {
        result = a + b;
        return result;
    }
    
    float divide(float x, float y) {
        if (y != 0) {
            return x / y;
        }
        return 0.0;
    }
};

int main() {
    int num1 = 10;
    int num2 = 20;
    float sum = 0.0;
    
    // Simple calculator program
    Calculator calc;
    sum = calc.add(num1, num2);
    
    if (sum > 25) {
        cout << "Sum is greater than 25";
    } else {
        cout << "Sum is less than or equal to 25";
    }
    
    for (int i = 0; i < 5; i++) {
        num1++;
    }
    
    while (num2 > 0) {
        num2--;
    }
    
    return 0;
}
